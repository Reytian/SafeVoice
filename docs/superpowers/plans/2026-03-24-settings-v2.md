# Settings V2 Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Make SafeVoice settings fully interactive — editable modes with style presets, vocabulary CRUD, and model selection with local/cloud LLM support.

**Architecture:** Three independent features layered onto the existing settings window. Style presets extend `modes.py`. LLM backend abstraction (`llm_backend.py`) replaces direct Ollama calls. Settings UI gets interactive controls (buttons, text fields, dropdowns) using existing PyObjC patterns. Cloud API support uses stdlib `urllib.request` — no new dependencies.

**Tech Stack:** Python 3.14, PyObjC (AppKit/Foundation), urllib.request, SQLite3

**Codebase conventions:**
- All instance attrs: `self._` prefix
- Python: `.venv/bin/python`
- Tests: `.venv/bin/python -m pytest tests/ -v`
- UI thread dispatch: `_ensure_main_thread` decorator or `performSelectorOnMainThread`
- NSObject trampolines for button callbacks: `_SettingsCallbackTarget` pattern
- Settings: `SettingsManager.get(key, default)` / `.set(key, value)`
- `LLMCleanup.cleanup(raw_text, languages=None, custom_prompt=None)`
- 1Password SSH signing may fail — use `git config --local commit.gpgsign false` before commit, unset after

```python
"""LLM backend abstraction for local (Ollama) and cloud (OpenAI/Anthropic/Google) APIs."""
import json
import logging
import urllib.request
import urllib.error
from typing import Optional

logger = logging.getLogger(__name__)

OLLAMA_BASE = "http://localhost:11434"


class LLMBackend:
    """Abstract base for LLM backends."""
    name: str = "Unknown"

    def chat(self, system_prompt: str, user_message: str, temperature: float = 0.0,
             max_tokens: int = 512) -> Optional[str]:
        raise NotImplementedError

    def is_available(self) -> bool:
        raise NotImplementedError

    def warm_up(self) -> None:
        pass


class OllamaBackend(LLMBackend):
    """Local Ollama backend."""

    def __init__(self, model: str = "qwen2.5:3b"):
        self.model = model
        self.name = f"Ollama ({model})"
        self._available: Optional[bool] = None

    def is_available(self) -> bool:
        if self._available is True:
            return True
        try:
            req = urllib.request.Request(f"{OLLAMA_BASE}/api/tags")
            with urllib.request.urlopen(req, timeout=2) as resp:
                data = json.loads(resp.read())
                names = [m["name"] for m in data.get("models", [])]
                self._available = (
                    self.model in names or f"{self.model}:latest" in names
                )
                return self._available
        except Exception:
            self._available = None
            return False

    def warm_up(self) -> None:
        if not self.is_available():
            return
        try:
            body = json.dumps({
                "model": self.model, "keep_alive": "30m", "prompt": "",
            }).encode()
            req = urllib.request.Request(
                f"{OLLAMA_BASE}/api/generate", data=body,
                headers={"Content-Type": "application/json"}, method="POST",
            )
            with urllib.request.urlopen(req, timeout=30):
                pass
            logger.info("Ollama model warmed up: %s", self.model)
        except Exception as e:
            logger.warning("Ollama warm-up failed: %s", e)

    def chat(self, system_prompt: str, user_message: str, temperature: float = 0.0,
             max_tokens: int = 512) -> Optional[str]:
        if not self.is_available():
            return None
        try:
            messages = [
                {"role": "system", "content": system_prompt},
                {"role": "user", "content": user_message},
            ]
            body = json.dumps({
                "model": self.model, "messages": messages, "stream": False,
                "options": {"temperature": temperature, "num_predict": max_tokens, "top_p": 0.9},
            }).encode()
            req = urllib.request.Request(
                f"{OLLAMA_BASE}/api/chat", data=body,
                headers={"Content-Type": "application/json"}, method="POST",
            )
            with urllib.request.urlopen(req, timeout=15) as resp:
                data = json.loads(resp.read())
                return data["message"]["content"].strip()
        except Exception as e:
            logger.warning("Ollama chat failed: %s", e)
            return None

    @staticmethod
    def list_models() -> list[dict]:
        """Return list of locally available Ollama models."""
        try:
            req = urllib.request.Request(f"{OLLAMA_BASE}/api/tags")
            with urllib.request.urlopen(req, timeout=3) as resp:
                data = json.loads(resp.read())
                return [
                    {"name": m["name"], "size": m.get("size", 0)}
                    for m in data.get("models", [])
                ]
        except Exception:
            return []


class CloudBackend(LLMBackend):
    """Cloud API backend (OpenAI, Anthropic, Google)."""

    def __init__(self, provider: str, model: str, api_key: str):
        self.provider = provider
        self.model = model
        self.api_key = api_key
        self.name = f"{provider.title()} ({model})"

    def is_available(self) -> bool:
        return bool(self.api_key)

    def _build_request(self, user_message: str, system_prompt: str) -> tuple[str, dict, bytes]:
        """Build (url, headers, body) for the provider."""
        if self.provider == "openai":
            url = "https://api.openai.com/v1/chat/completions"
            headers = {
                "Authorization": f"Bearer {self.api_key}",
                "Content-Type": "application/json",
            }
            body = json.dumps({
                "model": self.model,
                "messages": [
                    {"role": "system", "content": system_prompt},
                    {"role": "user", "content": user_message},
                ],
                "temperature": 0.0, "max_tokens": 512,
            }).encode()

        elif self.provider == "anthropic":
            url = "https://api.anthropic.com/v1/messages"
            headers = {
                "x-api-key": self.api_key,
                "anthropic-version": "2023-06-01",
                "Content-Type": "application/json",
            }
            body = json.dumps({
                "model": self.model, "max_tokens": 512,
                "system": system_prompt,
                "messages": [{"role": "user", "content": user_message}],
            }).encode()

        elif self.provider == "google":
            url = (
                f"https://generativelanguage.googleapis.com/v1beta/"
                f"models/{self.model}:generateContent?key={self.api_key}"
            )
            headers = {"Content-Type": "application/json"}
            body = json.dumps({
                "system_instruction": {"parts": [{"text": system_prompt}]},
                "contents": [{"parts": [{"text": user_message}]}],
                "generationConfig": {"temperature": 0.0, "maxOutputTokens": 512},
            }).encode()
        else:
            raise ValueError(f"Unknown provider: {self.provider}")

        return url, headers, body

    def chat(self, system_prompt: str, user_message: str, temperature: float = 0.0,
             max_tokens: int = 512) -> Optional[str]:
        if not self.is_available():
            return None
        try:
            url, headers, body = self._build_request(user_message, system_prompt)
            req = urllib.request.Request(url, data=body, headers=headers, method="POST")
            with urllib.request.urlopen(req, timeout=15) as resp:
                data = json.loads(resp.read())

            # Extract text from provider-specific response format
            if self.provider == "openai":
                return data["choices"][0]["message"]["content"].strip()
            elif self.provider == "anthropic":
                return data["content"][0]["text"].strip()
            elif self.provider == "google":
                return data["candidates"][0]["content"]["parts"][0]["text"].strip()
        except Exception as e:
            logger.warning("Cloud LLM (%s) failed: %s", self.provider, e)
            return None
```

```python
def _make_vocab_delete_target(self, kind, key):
    """Create callback target for vocabulary delete buttons."""
    def _delete():
        if kind == "hotword":
            self._vocabulary_manager.remove_hotword(key)
        else:
            self._vocabulary_manager.remove_snippet(key)
        self._refresh_vocabulary_tab()

    target = _SettingsCallbackTarget.alloc().initWithCallback_(_delete)
    self._callback_prevent_gc.add(target)
    return target

def _make_vocab_add_target(self, kind):
    """Create callback target for vocabulary add buttons."""
    def _add():
        if kind == "hotword":
            word = self._hw_field.stringValue().strip()
            if word:
                self._vocabulary_manager.add_hotword(word)
                self._hw_field.setStringValue_("")
        else:
            trigger = self._sn_trigger_field.stringValue().strip()
            replacement = self._sn_replacement_field.stringValue().strip()
            if trigger and replacement:
                self._vocabulary_manager.add_snippet(trigger, replacement)
                self._sn_trigger_field.setStringValue_("")
                self._sn_replacement_field.setStringValue_("")
        self._refresh_vocabulary_tab()

    target = _SettingsCallbackTarget.alloc().initWithCallback_(_add)
    self._callback_prevent_gc.add(target)
    return target

def _refresh_vocabulary_tab(self):
    """Rebuild vocabulary tab content after changes."""
    if hasattr(self, '_vocab_container') and self._vocab_container:
        self._populate_vocabulary_view(self._vocab_container)
```

```python
def _load_api_key(self, field, provider):
    """Load API key from credentials file."""
    import os
    cred_path = os.path.expanduser("~/.config/safevoice/credentials.json")
    try:
        if os.path.exists(cred_path):
            import json
            with open(cred_path) as f:
                creds = json.load(f)
            key = creds.get(provider, "")
            if key:
                field.setStringValue_(key)
    except Exception:
        pass

def _save_api_key(self, provider, api_key):
    """Save API key to credentials file with 0600 permissions."""
    import os, json
    cred_path = os.path.expanduser("~/.config/safevoice/credentials.json")
    creds = {}
    try:
        if os.path.exists(cred_path):
            with open(cred_path) as f:
                creds = json.load(f)
    except Exception:
        pass
    creds[provider] = api_key
    os.makedirs(os.path.dirname(cred_path), exist_ok=True)
    with open(cred_path, "w") as f:
        json.dump(creds, f, indent=2)
    os.chmod(cred_path, 0o600)
```
